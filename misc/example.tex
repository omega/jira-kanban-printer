\slide{Some options}

\begin{list1}
\item StarOffice
	\begin{list2}
	\item Poor font support
	\item Cumbersome GUI 
	\item Must have StarOffice to display presentation
		\begin{list3}
		\item PowerPoint support not perfect
		\end{list3}
	\end{list2}

\item MagicPoint
	\begin{list2}
	\item I have used this for a long time...
	\item But a little too limited
	\item Hard to add new features
	\end{list2}
\end{list1}

\slide{The Solution: PDFLaTeX}

\begin{list1}
\item Version of LaTeX which produces PDF directly 
	\begin{list2}
	\item Advanced layout properties
	\item Great graphics and figure support
	\item Works with TrueType fonts
	\end{list2}

\item Why PDF?
	\begin{list2}
	\item Almost ubiquitous
	\item Always looks the same
	\item Acroread {\blueem full screen mode} for display
	\end{list2}
\end{list1}
